用户选择，且使用老方式选择用户

HTML 
  <input data-type="names"  data-dialog=true data-hasbt=true   name="name2" value="李小靓" />
js: 
          $("input[name=name2]").on("select load",function(e,item){
                if(!item) return false;
                var res = []
                $.each(item,function(name,value){
                      res.push(name+":"+value)
                })
                $("#name2p").html(res.join("<br>"))
            })
